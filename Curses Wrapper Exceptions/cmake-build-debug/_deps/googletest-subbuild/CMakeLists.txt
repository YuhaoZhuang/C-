cmake_minimum_required(VERSION 3.13.2)

# We name the project and the target for the ExternalProject_Add() call
# to something that will highlight to the user what we are working on if
# something goes wrong and an error message is produced.

project(googletest-populate NONE)

include(ExternalProject)
ExternalProject_Add(googletest-populate
                     "UPDATE_DISCONNECTED" "False" "GIT_REPOSITORY" "https://github.com/google/googletest.git" "GIT_TAG" "master"
                    SOURCE_DIR          "/cygdrive/c/Curses Wrapper Exceptions/cmake-build-debug/_deps/googletest-src"
                    BINARY_DIR          "/cygdrive/c/Curses Wrapper Exceptions/cmake-build-debug/_deps/googletest-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
)
